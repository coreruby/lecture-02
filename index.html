Method

Defined by the keyword def 

def eat(food)
  3.times { nom }
end

Method

Returns a value 

def eat(food)
  3.times { nom }
  return :delicious
end

>> eat(:stake)
=> :delicious

Method

Returns a value early 

def eat(food)
  return :oh_well if food == :potato

  3.times { nom }
  return :delicious
end

>> eat(:stake)
=> :delicious

Method

Returns a value implicitly 

def eat(food)
  3.times { nom }
  :delicious
end

>> eat(:stake)
=> :delicious

Method

Always returns a value 

def nothing
end

>> nothing
=> nil

Method

Can take many arguments 

def eat(meal1, meal2, meal3)
  nom meal1
  nom meal2
  nom meal3
end

Method

Can take any arguments 

def eat(*meals)
  meals.each { |meal| nom meal }
end
